# expectJsonLength

Asserts on the length of JSON array objects.

# Syntax

expectJsonLength(length)
expectJsonLength(path, length)
1
2

# Usage

# ✅ Correct Usage

await spec()
  .get('/api/users')
  .expectJsonLength(1);
1
2
3
// bdd style
const _spec = await spec().get('/api/users');
_spec.response().to.have.jsonLength(1);
1
2
3

# Arguments

# > length (number)

Expected json object length.

# > path (string)

Json path. See json-query (opens new window) for more usage details.

# Examples

# Normal

const { spec } = require('pactum');

await spec()
  .get('https://jsonplaceholder.typicode.com/users')
  .expectJsonLength(10);
